function crc(bytes) {
	return bytes
		.reduce((bits, byte) => bits.concat([7, 6, 5, 4, 3, 2, 1, 0].map(i => byte >> i & 1)), [])
		.reduce((rem, bit) => ((rem << 1) + (rem >> 7) ^ bit) % 256, 255)
}
